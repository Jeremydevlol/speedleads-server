## üîß Configuraci√≥n de Axios/Fetch

### Opci√≥n 1: Axios (Recomendado)

```javascript
// lib/axios.js o config/api.js
import axios from 'axios';

const api = axios.create({
  baseURL: process.env.NEXT_PUBLIC_API_URL || 'https://speedleads-server.onrender.com',
  timeout: 30000, // 30 segundos
  withCredentials: true, // IMPORTANTE: Para cookies/sesiones
  headers: {
    'Content-Type': 'application/json',
  }
});

// Interceptor para agregar token si existe
api.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// Interceptor para manejar errores
api.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      // Redirigir a login si no est√° autenticado
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default api;
```

### Opci√≥n 2: Fetch Nativo

```javascript
// lib/api.js
const API_URL = process.env.NEXT_PUBLIC_API_URL || 'https://speedleads-server.onrender.com';

export async function fetchAPI(endpoint, options = {}) {
  const token = localStorage.getItem('token');
  
  const response = await fetch(`${API_URL}${endpoint}`, {
    ...options,
    credentials: 'include', // IMPORTANTE: Para cookies
    headers: {
      'Content-Type': 'application/json',
      ...(token && { Authorization: `Bearer ${token}` }),
      ...options.headers,
    },
  });

  if (!response.ok) {
    if (response.status === 401) {
      window.location.href = '/login';
    }
    throw new Error(`API Error: ${response.status}`);
  }

  return response.json();
}
```

## üîå Configuraci√≥n de Socket.IO

```javascript
// lib/socket.js
import { io } from 'socket.io-client';

const SOCKET_URL = process.env.NEXT_PUBLIC_WS_URL || 'https://speedleads-server.onrender.com';

export const socket = io(SOCKET_URL, {
  withCredentials: true, // IMPORTANTE: Para autenticaci√≥n
  transports: ['websocket', 'polling'], // Intentar WebSocket primero
  autoConnect: false, // Conectar manualmente cuando sea necesario
  reconnection: true,
  reconnectionDelay: 1000,
  reconnectionAttempts: 5,
});

// Conectar cuando el usuario est√© autenticado
export function connectSocket(userId) {
  if (!socket.connected) {
    socket.auth = { userId };
    socket.connect();
  }
}

// Desconectar cuando el usuario cierre sesi√≥n
export function disconnectSocket() {
  if (socket.connected) {
    socket.disconnect();
  }
}

// Event listeners
socket.on('connect', () => {
  console.log('‚úÖ Socket conectado:', socket.id);
});

socket.on('disconnect', () => {
  console.log('‚ùå Socket desconectado');
});

socket.on('error', (error) => {
  console.error('‚ùå Socket error:', error);
});

export default socket;
```

  );

## üö® Troubleshooting

### Error: CORS
**Problema**: `Access to fetch at 'https://speedleads-server.onrender.com' from origin 'https://app.uniclick.io' has been blocked by CORS`

**Soluci√≥n**: El backend ya est√° configurado para permitir `https://app.uniclick.io`. Aseg√∫rate de:
1. Usar `credentials: 'include'` o `withCredentials: true`
2. No usar `localhost` en producci√≥n
3. Verificar que el origen sea exactamente `https://app.uniclick.io`

### Error: 401 Unauthorized
**Problema**: Requests fallan con 401

**Soluci√≥n**:
1. Verificar que el token est√© en localStorage
2. Agregar header `Authorization: Bearer ${token}`
3. Verificar que `withCredentials: true` est√© configurado

### Error: Socket no conecta
**Problema**: Socket.IO no se conecta

**Soluci√≥n**:
1. Usar `wss://` en producci√≥n (no `ws://`)
2. Agregar `withCredentials: true`
3. Verificar que el backend est√© corriendo
